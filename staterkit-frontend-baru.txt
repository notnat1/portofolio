// client/src/App.jsx (Contoh dengan Tailwind)
import React, { useState, useEffect } from 'react';
import axios from 'axios';
import io from 'socket.io-client'; // Untuk Live Counter

// URL Backend lo (akan diganti ke domain Contabo nanti)
const API_URL = 'http://localhost:3001/api';

const socket = io('http://localhost:3001'); // Koneksi Socket.io

const App = () => {
    const [isDarkMode, setIsDarkMode] = useState(true);
    const [onlineCount, setOnlineCount] = useState(0);
    const [projects, setProjects] = useState([]);

    // --- EFFECT 1: REAL-TIME COUNTER ---
    useEffect(() => {
        socket.on('userCount', (count) => {
            setOnlineCount(count);
        });
        
        // --- EFFECT 2: FETCH PROJECTS DARI BACKEND ---
        const fetchProjects = async () => {
            try {
                // Endpoint baru lo dari database PostgreSQL
                const response = await axios.get(`${API_URL}/projects`); 
                setProjects(response.data);
            } catch (error) {
                console.error("Gagal fetch data:", error);
            }
        };

        fetchProjects();
        
        return () => socket.off('userCount');
    }, []);

    const bgColor = isDarkMode ? 'bg-gray-900' : 'bg-white';
    const textColor = isDarkMode ? 'text-gray-50' : 'text-gray-800';

    return (
        <div className={`min-h-screen ${bgColor} ${textColor} transition duration-500`}>
            {/* HEADER & PROFILE SECTION */}
            <header className="fixed w-full z-10 p-4 bg-gray-800 shadow-lg">
                <div className="flex justify-between items-center max-w-6xl mx-auto">
                    <h1 className="text-2xl font-extrabold text-blue-400">Nat.dev</h1>
                    {/* Dark Mode Toggle - Cuma buat keren-kerenan */}
                    <button onClick={() => setIsDarkMode(!isDarkMode)} className="p-2 rounded-full hover:bg-gray-700">
                        {isDarkMode ? 'ðŸŒž' : 'ðŸŒ™'}
                    </button>
                </div>
            </header>

            <main className="max-w-6xl mx-auto pt-20 p-4">
                {/* LIVE COUNTER (REAL-TIME PROOF) */}
                <div className="bg-green-600 p-3 rounded-lg text-white text-center mb-10 shadow-xl">
                    ðŸŸ¢ {onlineCount} Users Currently Viewing My Portfolio (Real-Time Proof via Socket.io)
                </div>

                {/* 1. SKILLS & DEVOPS SECTION */}
                <section id="skills" className="py-12 border-b border-gray-700">
                    <h2 className="text-4xl font-bold mb-8 text-center text-blue-400">My T-Shaped Expertise</h2>
                    {/* Nanti diisi grid: Kiri (Fullstack), Kanan (DevOps/Cloud) */}
                    <div className="grid grid-cols-1 md:grid-cols-2 gap-8">
                        <div className="p-6 bg-gray-800 rounded-xl shadow-2xl">
                            <h3 className="text-2xl font-semibold mb-4 text-orange-400">Platform & DevOps (My Edge)</h3>
                            <ul className="list-disc list-inside text-gray-300">
                                <li>Cloud Infrastructure as Code (**Terraform/Ansible**)</li>
                                <li>AWS/GCP Deployment & Management</li>
                                <li>Containerization (**Docker**) and Orchestration (K8s concept)</li>
                                <li>CI/CD Automation (**GitHub Actions/Jenkins**)</li>
                            </ul>
                        </div>
                        <div className="p-6 bg-gray-800 rounded-xl shadow-2xl">
                            <h3 className="text-2xl font-semibold mb-4 text-teal-400">Fullstack Development</h3>
                            <ul className="list-disc list-inside text-gray-300">
                                <li>Frontend: **React/Typescript, Angular, Vue.js**</li>
                                <li>Backend: **Node.js (Express), Python (FastAPI/Django)**</li>
                                <li>Database: **PostgreSQL**, MySQL, MongoDB</li>
                                <li>Testing: Unit & Integration Testing (Vitest/Jest)</li>
                            </ul>
                        </div>
                    </div>
                </section>

                {/* 2. PROJECTS SECTION */}
                <section id="projects" className="py-12 border-b border-gray-700">
                    <h2 className="text-4xl font-bold mb-8 text-center text-blue-400">Mission-Critical Projects ({projects.length} Found)</h2>
                    {/* Ini nanti lo isi dengan data 'projects' dari PostgreSQL */}
                    {/* Lo bisa pakai data dari Inventaris lo di sini, atau data PollStream */}
                </section>

                {/* 3. CONTACT FORM SECTION */}
                <section id="contact" className="py-12">
                    <h2 className="text-4xl font-bold mb-8 text-center text-blue-400">Get In Touch (Backend Functional Proof)</h2>
                    {/* Form sederhana yang terhubung ke Backend Node.js/PostgreSQL */}
                    {/* Nanti tinggal lo pasang handleContactSubmit dari Starter Kit sebelumnya */}
                </section>
            </main>
            <footer className="text-center py-4 text-gray-500 text-sm">
                Built by Natte with React, Node.js, and a lot of PostgreSQL.
            </footer>
        </div>
    );
};

export default App;
